--账号表
create table sys_account(
    "id" int primary key identity,
    "account"  varchar(16) not null,
    "password"  varchar(16) not null,
    "salf" char(8) not null,
    "explain"  text default null
);

    "id" int primary key identity,
    "role"  varchar(16) not null,
    "explain"  text default null
);
